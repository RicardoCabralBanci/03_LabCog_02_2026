# Código VBA Final (Versão 3.1 - Grid 3x & Clean UI)

**Instruções de Uso**:
1. Copie todo o bloco de código abaixo.
2. Cole em um Módulo VBA (Substitua qualquer versão anterior).
3. Certifique-se de ter a pasta `\Imagens\` ao lado do seu arquivo Excel.

```vba
Attribute VB_Name = "mod_UI_Core"
Option Explicit

' ==============================================================================
' ARQUITETURA DE UI - VERSÃO 3.1 (MATRIX GRID & RELATIVE PATHS)
' ==============================================================================

' --- GEOMETRIA ---
Private Const COL_SIDEBAR As String = "A"
Private Const COL_GUTTER As String = "B"
Private Const ROW_START_UI As Integer = 5     ' Start na Linha 5
Private Const BTN_HEIGHT_CELLS As Integer = 2 ' Botão = 2 células de altura
Private Const BTN_MARGIN_H As Double = 2      ' Margem lateral mínima
Private Const CARD_HEIGHT_CELLS As Integer = 4 ' Card = 4 células de altura

' --- CORES ---
Private Const CLR_SIDEBAR_BG As Long = 3618615   ' Cinza Chumbo
Private Const CLR_TEXT_WHITE As Long = 16777215
Private Const CLR_TEXT_GREY As Long = 8421504
Private Const CLR_CARD_BG As Long = 16316664     ' Cinza Claro (RGB 248)

' Cores dos Filtros (Estado Clean)
Private Const CLR_BTN_ACTIVE As Long = 2631720   ' Cinza Escuro (Grafite)
Private Const CLR_BTN_INACTIVE As Long = 14277081 ' Cinza Suave (Desativado)

' Cores dos Badges (Marcas)
Private Const CLR_BRAND_KHS As Long = 9863790    ' Clearline
Private Const CLR_BRAND_SIE As Long = 12411904   ' Siemens
Private Const CLR_BRAND_ROC As Long = 46116      ' Rockwell

' --- DADOS ---
Public Type T_IHM_Card
    ID As Long
    Fabricante As String
    Modelo As String
    Tamanho As String
    Engine As String
    ImgPath As String
End Type

Private Catalog() As T_IHM_Card
Private CatalogCount As Integer
Private IsCatalogLoaded As Boolean

' --- FILTROS ---
Public Filter_Clearline As Boolean
Public Filter_Siemens As Boolean
Public Filter_Rockwell As Boolean

' ==============================================================================
' 1. MAIN SETUP
' ==============================================================================
Public Sub Init_UI()
    Dim ws As Worksheet
    Set ws = ActiveSheet
    
    Application.ScreenUpdating = False
    
    ' Carrega Dados e Limpa
    Load_Catalog_From_DB
    
    On Error Resume Next
    ws.DrawingObjects.Delete
    ws.Cells.Interior.Color = xlNone
    ws.Rows.RowHeight = 15 ' Reset Altura Padrão
    On Error GoTo 0
    
    ' Configura Colunas
    ws.Columns(COL_SIDEBAR).ColumnWidth = 25
    ' Pinta Sidebar (Apenas L4 em diante)
    ws.Range(ws.Cells(4, 1), ws.Cells(ws.Rows.Count, 1)).Interior.Color = CLR_SIDEBAR_BG
    
    ws.Columns(COL_GUTTER).ColumnWidth = 2
    ws.Columns("C:E").ColumnWidth = 22 ' Grid 3 Colunas
    
    ' Inicializa Filtros
    If Not Filter_Clearline And Not Filter_Siemens And Not Filter_Rockwell Then
        Filter_Clearline = True: Filter_Siemens = True: Filter_Rockwell = True
    End If
    
    Render_Sidebar
    Render_Gallery_Matrix
    
    ws.Range("A1").Select
    Application.ScreenUpdating = True
End Sub

' ==============================================================================
' 2. DATA LOADER
' ==============================================================================
Private Sub Load_Catalog_From_DB()
    Dim wsDB As Worksheet, lastRow As Long, i As Long
    On Error Resume Next
    Set wsDB = Planilha66
    If wsDB Is Nothing Then Set wsDB = ThisWorkbook.Sheets("ihm_models")
    On Error GoTo 0
    
    If wsDB Is Nothing Then MsgBox "Erro: Planilha ihm_models não encontrada.": Exit Sub
    
    lastRow = wsDB.Cells(wsDB.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub
    
    CatalogCount = lastRow - 1
    ReDim Catalog(1 To CatalogCount)
    
    For i = 2 To lastRow
        With Catalog(i - 1)
            .ID = wsDB.Cells(i, 1).Value
            .Fabricante = wsDB.Cells(i, 3).Value
            .Modelo = wsDB.Cells(i, 4).Value
            .Tamanho = wsDB.Cells(i, 5).Value
            .Engine = wsDB.Cells(i, 6).Value
            .ImgPath = wsDB.Cells(i, 8).Value
        End With
    Next i
    IsCatalogLoaded = True
End Sub

' ==============================================================================
' 3. SIDEBAR RENDERER (CLEAN UI)
' ==============================================================================
Private Sub Render_Sidebar()
    Dim ws As Worksheet: Set ws = ActiveSheet
    Dim currRow As Integer
    
    ' Cabeçalho
    With ws.Cells(ROW_START_UI - 1, 1)
        .Value = "FILTROS"
        .Font.Color = CLR_TEXT_GREY: .Font.Bold = True: .HorizontalAlignment = xlCenter
        .Font.Size = 8
    End With
    
    currRow = ROW_START_UI
    
    ' Botões (Gap = 0, Feedback por Cor)
    Draw_Sidebar_Btn "btn_filter_khs", "CLEARLINE", currRow, Filter_Clearline
    currRow = currRow + BTN_HEIGHT_CELLS
    
    Draw_Sidebar_Btn "btn_filter_sie", "SIEMENS", currRow, Filter_Siemens
    currRow = currRow + BTN_HEIGHT_CELLS
    
    Draw_Sidebar_Btn "btn_filter_roc", "ROCKWELL", currRow, Filter_Rockwell
End Sub

Private Sub Draw_Sidebar_Btn(id As String, label As String, startRow As Integer, isOn As Boolean)
    Dim ws As Worksheet: Set ws = ActiveSheet
    Dim rng As Range, btn As Shape
    
    Set rng = ws.Range(ws.Cells(startRow, 1), ws.Cells(startRow + BTN_HEIGHT_CELLS - 1, 1))
    
    ' Botão com margem vertical interna simulada
    Set btn = ws.Shapes.AddShape(msoShapeRoundedRectangle, _
        rng.Left + BTN_MARGIN_H, rng.Top + 2, rng.Width - (BTN_MARGIN_H * 2), rng.Height - 4)
        
    With btn
        .Name = id
        .TextFrame.Characters.Text = label
        .TextFrame.Characters.Font.Bold = True: .TextFrame.Characters.Font.Size = 9
        .Line.Visible = msoFalse: .OnAction = "Toggle_Filter_Handler"
        
        If isOn Then
            .Fill.ForeColor.RGB = CLR_BTN_ACTIVE ' Grafite
            .TextFrame.Characters.Font.Color = CLR_TEXT_WHITE
            .Shadow.Type = msoShadow25: .Shadow.Transparency = 0.8
        Else
            .Fill.ForeColor.RGB = CLR_BTN_INACTIVE ' Cinza Suave
            .TextFrame.Characters.Font.Color = RGB(100, 100, 100)
            .Fill.Transparency = 0 ' Opaco para cobrir fundo
        End If
    End With
End Sub

' ==============================================================================
' 4. GALLERY RENDERER (MATRIX 3X & RELATIVE PATHS)
' ==============================================================================
Public Sub Render_Gallery_Matrix()
    Dim ws As Worksheet: Set ws = ActiveSheet
    Dim i As Integer, counter As Integer
    Dim targetCol As Integer, targetRow As Integer, showItem As Boolean
    
    If Not IsCatalogLoaded Then Load_Catalog_From_DB
    
    ' Limpa Galeria
    Dim s As Shape
    For Each s In ws.Shapes
        If s.Name Like "grp_card_*" Or s.Name = "header_catalog" Then s.Delete
    Next s
    
    Draw_Gallery_Header
    
    counter = 0
    targetRow = ROW_START_UI
    
    For i = 1 To CatalogCount
        showItem = False
        Select Case UCase(Catalog(i).Fabricante)
            Case "CLEARLINE": If Filter_Clearline Then showItem = True
            Case "SIEMENS":   If Filter_Siemens Then showItem = True
            Case "ROCKWELL":  If Filter_Rockwell Then showItem = True
        End Select
        
        If showItem Then
            ' Matrix: Colunas 3, 4, 5 (C, D, E)
            targetCol = 3 + (counter Mod 3)
            
            ' Desce linha a cada 3
            If counter > 0 And (counter Mod 3) = 0 Then
                targetRow = targetRow + CARD_HEIGHT_CELLS
            End If
            
            Draw_Card_Matrix Catalog(i), targetRow, targetCol
            counter = counter + 1
        End If
    Next i
End Sub

Private Sub Draw_Gallery_Header()
    Dim ws As Worksheet: Set ws = ActiveSheet
    Dim rng As Range, shp As Shape
    
    Set rng = ws.Range("C4:E4")
    Set shp = ws.Shapes.AddShape(msoShapeRectangle, rng.Left, rng.Top, rng.Width, rng.Height)
    With shp
        .Name = "header_catalog"
        .Fill.ForeColor.RGB = RGB(60, 60, 60)
        .Line.Visible = msoFalse
        .TextFrame.Characters.Text = "CATÁLOGO DE MODELOS"
        .TextFrame.Characters.Font.Color = vbWhite
        .TextFrame.Characters.Font.Bold = True
        .TextFrame.Characters.Font.Size = 9
    End With
End Sub

Private Sub Draw_Card_Matrix(data As T_IHM_Card, rowIdx As Integer, colIdx As Integer)
    Dim ws As Worksheet: Set ws = ActiveSheet
    Dim rng As Range, bg As Shape, img As Shape, txt As Shape
    Dim shpNames() As String: Dim shpCount As Integer: shpCount = 0
    Dim margin As Double: margin = 4
    
    Set rng = ws.Range(ws.Cells(rowIdx, colIdx), ws.Cells(rowIdx + CARD_HEIGHT_CELLS - 1, colIdx))
    
    ' 1. Fundo
    Set bg = ws.Shapes.AddShape(msoShapeRoundedRectangle, _
        rng.Left + margin, rng.Top + margin, rng.Width - (margin * 2), rng.Height - (margin * 2))
    bg.Name = "tmp_bg_" & data.ID: bg.Fill.ForeColor.RGB = CLR_CARD_BG
    bg.Line.Visible = msoFalse: bg.Shadow.Type = msoShadow25: bg.Shadow.Transparency = 0.8
    AddShapeToArray bg.Name, shpNames, shpCount
    
    ' 2. Imagem (Lógica Relativa)
    Dim imgPath As String, imgDefault As String
    imgDefault = ThisWorkbook.path & "\Imagens\IHM_Generica.png"
    
    ' Define Caminho
    If data.ImgPath <> "" Then
        If InStr(data.ImgPath, ":") > 0 Then
            imgPath = data.ImgPath ' Absoluto
        Else
            imgPath = ThisWorkbook.path & "\" & data.ImgPath ' Relativo
        End If
    Else
        imgPath = imgDefault
    End If
    
    On Error Resume Next
    Set img = ws.Shapes.AddPicture(imgPath, msoFalse, msoTrue, bg.Left + 5, bg.Top + 5, 40, 40)
    If img Is Nothing Then
        ' Tenta Fallback Generico
        Set img = ws.Shapes.AddPicture(imgDefault, msoFalse, msoTrue, bg.Left + 5, bg.Top + 5, 40, 40)
    End If
    If img Is Nothing Then
        ' Fallback Final (X Vermelho)
        Set img = ws.Shapes.AddShape(msoShapeRectangle, bg.Left + 5, bg.Top + 5, 40, 40)
        img.Fill.ForeColor.RGB = vbWhite: img.Line.ForeColor.RGB = vbRed
    End If
    On Error GoTo 0
    
    img.Name = "tmp_img_" & data.ID
    AddShapeToArray img.Name, shpNames, shpCount
    
    ' 3. Título
    Set txt = ws.Shapes.AddTextbox(msoTextOrientationHorizontal, bg.Left + 50, bg.Top + 5, bg.Width - 55, 30)
    txt.Name = "tmp_txt_" & data.ID: txt.TextFrame.Characters.Text = data.Modelo
    txt.TextFrame.Characters.Font.Name = "Segoe UI": txt.TextFrame.Characters.Font.Bold = True
    txt.TextFrame.Characters.Font.Size = 9: txt.TextFrame.Characters.Font.Color = RGB(50, 50, 50)
    txt.Fill.Visible = msoFalse: txt.Line.Visible = msoFalse
    AddShapeToArray txt.Name, shpNames, shpCount
    
    ' 4. Badges
    Dim bColor As Long
    Select Case UCase(data.Fabricante)
        Case "SIEMENS": bColor = CLR_BRAND_SIE
        Case "CLEARLINE": bColor = CLR_BRAND_KHS
        Case "ROCKWELL": bColor = CLR_BRAND_ROC
        Case Else: bColor = CLR_TEXT_GREY
    End Select
    
    Dim b1 As Shape
    Set b1 = ws.Shapes.AddShape(msoShapeRoundedRectangle, bg.Left + 50, bg.Top + 35, 60, 15)
    b1.Name = "tmp_b1_" & data.ID: b1.Fill.ForeColor.RGB = bColor: b1.Line.Visible = msoFalse
    b1.TextFrame.Characters.Text = data.Engine: b1.TextFrame.Characters.Font.Size = 7
    b1.TextFrame.Characters.Font.Color = vbWhite
    AddShapeToArray b1.Name, shpNames, shpCount
    
    ' 5. Agrupar
    Dim grp As Shape
    Set grp = ws.Shapes.Range(shpNames).Group
    grp.Name = "grp_card_" & data.ID
    grp.OnAction = "Card_Click_Handler"
End Sub

' --- HELPERS ---
Private Sub AddShapeToArray(name As String, arr() As String, ByRef count As Integer)
    count = count + 1: ReDim Preserve arr(1 To count): arr(count) = name
End Sub

Public Sub Toggle_Filter_Handler()
    Dim id As String: id = Application.Caller
    If id = "btn_filter_khs" Then Filter_Clearline = Not Filter_Clearline
    If id = "btn_filter_sie" Then Filter_Siemens = Not Filter_Siemens
    If id = "btn_filter_roc" Then Filter_Rockwell = Not Filter_Rockwell
    Render_Sidebar: Render_Gallery_Matrix
End Sub

Public Sub Card_Click_Handler()
    MsgBox "ID: " & Replace(Application.Caller, "grp_card_", ""), vbInformation
End Sub
```
