Attribute VB_Name = "mod_Ribbon_NewMachines"
Option Explicit

' ==========================================================================================
' MÓDULO: mod_Ribbon_NewMachines
' DESCRIÇÃO: Callbacks para a Ribbon das novas máquinas (PLT e PCK)
' DATA: 13/01/2026
' ==========================================================================================

' --- PALETIZADOR (PLT) -------------------------------------------------------------------

'Callback for MecPLT onAction
Sub Mec_PLT(control As IRibbonControl)
    On Error GoTo ErroNavegacao
    Planilha43.Activate ' PLT Estrutura
    Exit Sub
ErroNavegacao:
    MsgBox "Erro ao acessar a aba 'Planilha43'. Verifique se ela foi clonada corretamente.", vbCritical
End Sub

'Callback for DadosPLT onAction
Sub Dados_PLT(control As IRibbonControl)
    On Error Resume Next
    Planilha44.Activate ' PLT Dados
End Sub

'Callback for ElPLT onAction
Sub El_PLT(control As IRibbonControl)
    On Error Resume Next
    Planilha45.Activate ' PLT Elétrica
End Sub

'Callback for IHMPLT onAction
Sub IHM_PLT(control As IRibbonControl)
    On Error Resume Next
    Planilha46.Activate ' PLT IHM
End Sub

'Callback for GerPLT onAction
Sub Gerar_PLT_Click(control As IRibbonControl)
    ' Chama a Nova Engine V5
    On Error GoTo ErroGeral
    
    ' Validação simples antes de chamar
    If MsgBox("Deseja gerar o manual do Paletizador agora?", vbQuestion + vbYesNo, "Confirmar Geração") = vbYes Then
        Call mod_NewEngine_V5.Gerar_PLT
    End If
    Exit Sub
    
ErroGeral:
    MsgBox "Erro ao tentar iniciar a geração: " & Err.Description, vbCritical
End Sub

'Callback for GIHMPLT onAction (Atualização de IHM - Placeholder)
Sub GIMH_PLT(control As IRibbonControl)
    MsgBox "Funcionalidade de atualização automática de IHM ainda em desenvolvimento para PLT.", vbInformation
End Sub
